import React, { useState, useEffect } from 'react';
import { useNavigate, useSearchParams } from 'react-router-dom';
import '../style/ChatRoom.css';

function ChatRoom() {
  const [searchParams] = useSearchParams();
  const chatType = searchParams.get('type') || 'text';
  const [message, setMessage] = useState('');
  const [onlineCount] = useState(1);
  const [isSearching, setIsSearching] = useState(true);
  const navigate = useNavigate();

  const handleExit = () => {
    navigate('/chat');
  };

  const handleSendMessage = () => {
    if (message.trim()) {
      console.log('Sending message:', message);
      setMessage('');
    }
  };

  const handleKeyPress = (e) => {
    if (e.key === 'Enter' && !e.shiftKey) {
      e.preventDefault();
      handleSendMessage();
    }
  };

  const handleNewChat = () => {
    setIsSearching(true);
    setMessage('');
  };

  return (
    <div className="chat-room">
      <header className="chat-header">
        <button className="exit-btn" onClick={handleExit}>
          ← Exit
        </button>
        
        <div className="header-center">
          <img src="/assets/logo.png" alt="Logo" className="header-logo" />
          <div className="header-text">
            <h1 className="header-title">omegle.com</h1>
            <p className="header-tagline">Talk to strangers!</p>
          </div>
        </div>
        
        <div className="online-status">
          <span className="eye-icon">👁</span>
          <span className="online-count">{onlineCount} online</span>
        </div>
      </header>

      <div className="chat-content">
        {chatType === 'video' ? (
          <>
            <div className="video-section">
              <div className="stranger-video">
                <div className="video-placeholder">
                  {isSearching && <div className="loading-spinner"></div>}
                  <div className="user-preview"></div>
                </div>
                <div className="video-label desktop-only">Stranger</div>
              </div>
              
              <div className="user-video">
                <div className="video-placeholder">
                  {/* User's video preview */}
                </div>
                <div className="video-label desktop-only">You</div>
              </div>

            </div>

            <div className="chat-section">
              <div className="messages-container">
                {/* Chat messages will appear here */}
              </div>
            </div>
          </>
        ) : (
          <div className="text-chat-section">
            <div className="messages-container">
              {/* Messages will appear here */}
            </div>
          </div>
        )}
      </div>

      <div className="chat-input-section">
        <input
          type="text"
          className="message-input"
          placeholder="Type your message..."
          value={message}
          onChange={(e) => setMessage(e.target.value)}
          onKeyPress={handleKeyPress}
        />
        
        <button className="send-btn" onClick={handleSendMessage}>
          &gt;
        </button>
        
        <button className="new-btn" onClick={handleNewChat}>
          ● New
        </button>
      </div>
    </div>
  );
}

export default ChatRoom;
